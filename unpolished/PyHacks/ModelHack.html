def pairCountsByCountBySortedFraction(fraction):
    pair_counts_by_count = []
    top_pair_fraction_by_distance = topPairs(int(fraction * len(all_distances)))

        min_score = 256 - i
        count = 0
        for pair in top_pair_fraction_by_distance:
            if pair[0] < min_score:
                break
            count = count + 1
